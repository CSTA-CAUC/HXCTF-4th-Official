2、之后就是用百度搜索定位时，input明明有值，但不显示出来

很蛋疼。
同样，问了度娘和群里大神，不过没有结果，大家认为这不咋可能！无奈之下，上了stackoverflow。秀了一把我的Cnglish！
看起来像是某个元素覆盖在上面，试了一下。无果！ 于是，自己想了一个办法，先不执行百度地图的方法 ，当focus（click事件也可以）这个input时才去执行百度的搜索方法。所以算是完美的解决了吧！ 

$("#address").focus(function(event) {
   var ac = new BMap.Autocomplete(    //建立一个自动完成的对象
    {"input" : "address"
    ,"location" : map1
  });
  ac.addEventListener("onhighlight", function(e) {  //鼠标放在下拉列表上的事件
  var str = "";
    var _value = e.fromitem.value;
    var value = "";
    if (e.fromitem.index > -1) {
      value = _value.province +  _value.city +  _value.district +  _value.street +  _value.business;
    }
    str = "FromItem<br />index = " + e.fromitem.index + "<br />value = " + value;

    value = "";
    if (e.toitem.index > -1) {
      _value = e.toitem.value;
      value = _value.province +  _value.city +  _value.district +  _value.street +  _value.business;
    }
    str += "<br />ToItem<br />index = " + e.toitem.index + "<br />value = " + value;
    G("searchResultPanel").innerHTML = str;

  });
  var myValue;
  ac.addEventListener("onconfirm", function(e) {    //鼠标点击下拉列表后的事件
  var _value = e.item.value;
    myValue = _value.province +  _value.city +  _value.district +  _value.street +  _value.business;
    G("searchResultPanel").innerHTML ="onconfirm<br />index = " + e.item.index + "<br />myValue = " + myValue;

    hAddress.val(myValue);
    setPlace();
  });
});
  function setPlace(){
    map1.clearOverlays();    //清除地图上所有覆盖物
    function myFun(){
      var pp = local.getResults().getPoi(0).point;    //获取第一个智能搜索的结果

       hLongitude.val(pp.lng);
      hLatitude.val(pp.lat);
      map1.centerAndZoom(pp, 18);
      map1.addOverlay(new BMap.Marker(pp));    //添加标注
    }
    var local = new BMap.LocalSearch(map1, { //智能搜索
      onSearchComplete: myFun
    });
    local.search(myValue);
  }

3、百度地图标注不在地图中心的解决方案

说起来也挺strange的，在不同的页面放置相同的百度地图组件(地图的js和css都是一模一样的)，结果是一个页面正常，另一个页面的标注不在地图的中间，而是处于地图container的左上角。 Baidu了一下，发现和我有同样问题的都有一个共同点，就是使用了tab选项切换卡。于是想起了关于浏览器解析的相关原理。 在开发中，tab2事先设置了display:none;，而设置这个属性之后，是不渲染出DOM结点，所以百度的JS无法获取到容器的宽高度，所以无法将标注标在地图的中间。而opacity是会渲染出DOM结点，所以改正之后，就不会出现这种问题了。

总的来说吧，百度地图比高德要好上手些，api简洁，但地图的详细度不如高德。
