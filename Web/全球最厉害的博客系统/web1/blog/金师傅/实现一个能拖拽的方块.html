实现一个能拖拽的方块
  • Kylin King
  • 二月 20 2017
  • 前端

看到一面试题，让实现一个能拖拽的方块。于是就写了一个很简易的 Demo

<div id="boxA" class="boxA" draggable="true">快来拖拽我啊！</div>  

#boxA {
  background-color: #832938;
  -webkit-user-select: none;
  -ms-user-select: none;
  -moz-user-select: none;
  padding: 10px;
  margin: 10px;
  width: 100px;
  height: 100px;
  box-sizing: border-box;
  left: 56px;
  top: 65px;
  cursor: pointer;
  position: absolute;
}

    var oBoxA = document.getElementById('boxA');

    var oLeft,oTop,initLeft,initTop;
    oBoxA.ondragstart = function (event) {
        var e = event || window.event;
        oLeft = e.clientX;
        oTop = e.clientY;
        initLeft = e.target.offsetLeft;
        initTop = e.target.offsetTop;

    };

    oBoxA.ondragover = function (event) {
        var e = event || window.event;
        var offsetX = e.clientX - oLeft;
        var offsetY = e.clientY - oTop;


        e.target.style.top = initTop+offsetY-10+ 'px';
        e.target.style.left = initLeft+offsetX-10 + 'px';

    };

    oBoxA.ondragleave = function (event){
        var e = event || window.event;
        var left = e.target.offsetLeft;
        var top = e.target.offsetTop;

        if(left < 40){
            e.target.style.left = '-10px';      
        }else if(top < 40){
            e.target.style.top = '-10px';
        }

    };
