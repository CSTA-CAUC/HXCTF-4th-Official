js实现中央播报台时间播报
  • Kylin King
  • 二月 22 2017
  • 前端

看到一道面试题：将现在的时间变成类似二〇一七年 二月 二十二日 星期三 十四时 三十三分 四十二秒这种大写的。于是写了一下：

<span id="time"></span>  

var NUM = ['〇', '一', '二', '三', '四', '五', '六', '七', '八', '九', '十'];  
var oSpan = document.getElementById('time');  
setInterval(function(){  
    var date = new Date();
    var nowTime = [date.getFullYear(),date.getMonth()+1,date.getDate(),date.getDay(),date.getHours(),date.getMinutes(),date.getSeconds()];
    var unit = ['年','月','日','星期','时','分','秒'];
    var str = '';

    for (var i = 0; i < nowTime.length; i++) {
        nowTime[i] = ('' + nowTime[i]).split('').map(function(value,index,array){

            for(var j=0;j<NUM.length;j++) {
                if(j == value){
                    // 去掉10,20,30 40 ，后面的〇
                    if(array[0] + array[1] >= 10 && i != 0 && array[1] == 0 && index != 0){
                        continue;
                    }
                    // 处理20,60
                    if(array[0] + array[1] >=20 && index == 0 && i != 0){
                        return NUM[j]+'十';
                    }else if(array[0]+array[1] >= 10 && array[0] + array[1] < 20 && index == 0){
                        // 10 - 19
                        return '十';
                    }else{
                        return NUM[j];
                    }

                }

            }
        }).join("");

        // 处理星期
        if(i != 3){
            nowTime[i] = nowTime[i]+unit[i];
        }else{
            nowTime[i] = unit[i]+nowTime[i];
        }
    };


    for(var i=0;i<nowTime.length;i++){
        str = str + nowTime[i] +' ';
    }

    console.log(nowTime);
    oSpan.innerHTML = str;
},1000);
