args = ['zip', '-r',

if os.path.isfile(os.path.join(basedir, 'build.json')):
    b = json.loads(
        open(os.path.join(basedir, 'build.json')).read())

        args.append(x)

        args.append("-x")
        args.append(x)
    p = subprocess.Popen(args, cwd=basedir)
